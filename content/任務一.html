任務一

# 用於讀取文件和處理路徑的庫
import os

# 2a.txt 文件的路徑
file_path = "2a.txt"

# 開啟文件並讀取每一行資料
with open(file_path, 'r') as file:
    lines = file.readlines()

# 遍歷每一行資料
for line in lines:
    # 分割每行資料並得到學號、GitHub 帳號和組別
    student_number, github_account, group = line.strip().split(',')

    # 生成學員倉庫和網站連結
    repository_link = f"https://github.com/{github_account}/cd2024"
    website_link = f"https://{github_account}.github.io/cd2024"

    # 生成 HTML 頁面的內容
    html_content = f"""
    <html>
    <head>
        <title>學員資訊</title>
    </head>
    <body>
        <h1>學員基本資訊</h1>
        <p>學號：{student_number}</p>
        <p>GitHub 帳號：{github_account}</p>
        <p>組別：{group}</p>
        <p>倉庫連結：<a href="{repository_link}">{repository_link}</a></p>
        <p>網站連結：<a href="{website_link}">{website_link}</a></p>
    </body>
    </html>
    """

    # 將 HTML 內容寫入文件
    output_file_name = f"{student_number}_info.html"
    with open(output_file_name, 'w') as output_file:
        output_file.write(html_content)

    print(f"已生成 HTML 文件：{output_file_name}")

print("完成")
